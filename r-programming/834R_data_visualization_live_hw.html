1st plot

Density Plot of petal length for iris

#1st plot
#Use iris dataset
#Density Plot of petal length for iris
ggplot(data = iris) +
  aes(x = Petal.Length, fill = Species) +
  labs(title="Density of Petal length in 3 Iris Species",
       x = "Petal length (cm)",
       y = "Sample density") +
  geom_density(alpha = 0.5) +
  #force x tick mark to be 0.5, start from min to max petal length
  scale_x_continuous(breaks = seq(min(iris$Petal.Length), max(iris$Petal.Length), by = 0.5)) +
  theme_minimal()

Facet from the above plot

  scale_x_continuous(breaks = seq(min(iris$Petal.Length), max(iris$Petal.Length), by = 0.5)) +
  theme_minimal() +
  facet_wrap(~Species, nrow = 3) #add this line for facet

3rd plot

Same data (petal length) but with box plot, add jitter, show more raw datapoint.

#3rd plot
#Use iris dataset
#Same data (petal length) but with box plot, add jitter, show more raw datapoint.
ggplot(data = iris, mapping = aes(y = Petal.Length, x = Species, color = Species)) +
  labs(title="Density of Petal length in 3 Iris Species",
       y = "Petal length (cm)",
       x = "Species") +
  geom_boxplot(alpha = 0.5) +
  geom_jitter(position = position_jitter(0.2)) +
  scale_y_continuous(breaks = seq(min(iris$Petal.Length), max(iris$Petal.Length), by = 0.5))

4th plot

Scatter plot of petal length(x) and width(y)

#4th plot
#Scatter plot of petal length(x) and width(y)
ggplot(data = iris, mapping = aes(x = Petal.Length, y = Petal.Width)) +
  geom_point(aes(color = Species)) +
  geom_smooth(method = lm) +
  labs(title="Correlation between petal length and width",
       x = "Petal length (cm)",
       y = "Petal Width (cm)") +
  scale_x_continuous(breaks = seq(min(iris$Petal.Length), max(iris$Petal.Length), by = 0.5)) +
  scale_y_continuous(breaks = seq(min(iris$Petal.Width), max(iris$Petal.Width), by = 0.2))
## `geom_smooth()` using formula = 'y ~ x'

Plotting

#5th plot
#Use USArrest dataset
#This is the information from 50 US states in 1973
#Geoplot of Murders by state

#Mutate the USArrest dataset, create the state column in lowercase, save in crimes
crimes <- as_tibble(USArrests, rownames="state") %>% mutate(state = tolower(state))

ggplot(crimes, aes(map_id = state)) +
  geom_map(aes(fill = Murder), map = states_map) +
  expand_limits(x = states_map$long, y = states_map$lat) +
  coord_map("albers", 40, 100) +
  labs(title="Murder Rate in USA (per 100,000)",
       x = "longitude",
       y = "latitude ")
